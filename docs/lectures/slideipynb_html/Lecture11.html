import numpy as np
import matplotlib.pyplot as plt

# Define the ReLU and Leaky ReLU functions
def relu(x):
    return np.maximum(0, x)

def leaky_relu(x, alpha=0.01):
    return np.maximum(alpha*x, x)

# Generate a range of values from -10 to 10
x = np.linspace(-10, 10, 1000)

# Compute the ReLU and Leaky ReLU values
relu_values = relu(x)
leaky_relu_values = leaky_relu(x, .1)

# Define the sigmoid function
def sigmoid(x):
    return 1 / (1 + np.exp(-x))

# Compute the sigmoid values
sigmoid_values = sigmoid(x)

# Add the sigmoid function to the plot
plt.figure()
plt.plot(x, relu_values, label='ReLU')
plt.plot(x, leaky_relu_values, label='Leaky ReLU')
plt.plot(x, sigmoid_values, label='Sigmoid')
#plt.title('ReLU vs Leaky ReLU vs Sigmoid')
plt.xlabel('x')
plt.ylabel('Activation Function Output')
plt.legend()
plt.grid(True)
plt.show()

import numpy as np
import matplotlib.pyplot as plt

# Define the ReLU and Leaky ReLU functions
def relu(x):
    return np.maximum(0, x)

def leaky_relu(x, alpha=0.01):
    return np.maximum(alpha*x, x)

# Generate a range of values from -10 to 10
x = np.linspace(-10, 10, 1000)

# Compute the ReLU and Leaky ReLU values
relu_values = relu(x)
leaky_relu_values = leaky_relu(x, .1)

# Define the sigmoid function
def sigmoid(x):
    return 1 / (1 + np.exp(-x))

# Compute the sigmoid values
sigmoid_values = sigmoid(x)

# Define the ELU function
def elu(x, alpha=1.0):
    return np.where(x > 0, x, alpha * (np.exp(x) - 1))

# Compute the ELU values
elu_values = elu(x)

# Add the ELU function to the plot
plt.figure()
plt.plot(x, relu_values, label='ReLU')
plt.plot(x, leaky_relu_values, label='Leaky ReLU')
plt.plot(x, sigmoid_values, label='Sigmoid')
plt.plot(x, elu_values, label='ELU')
plt.xlabel('x')
plt.ylabel('Activation Function Output')
plt.legend()
plt.grid(True)
plt.show()

import pydot
import random
from IPython.display import SVG

# Create a new graph
graph = pydot.Dot(graph_type='digraph', rankdir = "BT")

# Create nodes for the input layer
input_layer = [pydot.Node('input{}'.format(i)) for i in range(5)]

# Create nodes for the hidden layers
hidden_layer1 = [pydot.Node('hidden1{}'.format(i)) for i in range(5)]
hidden_layer2 = [pydot.Node('hidden2{}'.format(i)) for i in range(5)]

# Create a node for the output layer
output_layer = [pydot.Node('output')]

# Add the nodes to the graph
for node in input_layer + hidden_layer1 + hidden_layer2 + output_layer:
    graph.add_node(node)

# Connect the nodes
for i in range(5):
    for j in range(5):
        graph.add_edge(pydot.Edge(input_layer[i], hidden_layer1[j]))
        graph.add_edge(pydot.Edge(hidden_layer1[i], hidden_layer2[j]))

for node in hidden_layer2:
    graph.add_edge(pydot.Edge(node, output_layer[0]))

# Display the graph
SVG(graph.create(prog='dot', format='svg'))

import pydot
import random
from IPython.display import SVG

# Create a new graph
graph = pydot.Dot(graph_type='digraph', rankdir = "BT")

# Create nodes for the input layer
input_layer = [pydot.Node('input{}'.format(i)) for i in range(5)]
