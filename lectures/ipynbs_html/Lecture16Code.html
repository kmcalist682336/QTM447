# Define a dataset to generate arithmetic input-output pairs
class ArithmeticDataset(Dataset):
    def __init__(self, num_samples=10000, max_digits=3):
        self.num_samples = num_samples
        self.max_digits = max_digits
        self.CHARS = "0123456789+-"
        self.char2idx = {char: idx for idx, char in enumerate(self.CHARS)}
        self.idx2char = {idx: char for char, idx in self.char2idx.items()}
        self.operators = ['+', '-']
        self.questions, self.answers = self.generate_data()  
    
    def generate_data(self):
        questions, answers = [], []
        operator_cycle = ['+', '-']  # Define cyclic order
        op_index = 0  # Track the current operator
        
        while len(questions) < self.num_samples:
            a = random.randint(1, 10**self.max_digits - 1)
            b = random.randint(1, 10**self.max_digits - 1)
            op = operator_cycle[op_index % 2]  # Select the next operator in the cycle

            # Perform the operation
            if op == '+': result = a + b
            elif op == '-': result = a - b
            elif op == '*': result = a * b
            elif op == '/': 
                if b == 0: continue  # Avoid division by zero
                result = a // b  # Integer division

            input_str = f"{a}{op}{b}"
            output_str = str(result)

            questions.append(input_str)
            answers.append(output_str)
            
            op_index += 1  # Move to the next operator
        
        return questions, answers
  
    
    def encode_string(self, s, max_len):
        """
        Encodes a string into a tensor of indices, with padding.
        """
        encoded = [self.char2idx[c] for c in s]  # Convert characters to indices
        encoded = [0] * (max_len - len(encoded)) + encoded  # Left-padding
        
        return torch.tensor(encoded, dtype=torch.long)
    
    def decode_string(self, tensor):
        """
        Converts a batch of tensors into human-readable strings.
        """
        batch_decoded = []
        for seq in tensor:
            decoded_str = "".join(self.idx2char[idx.item()] for idx in seq if idx.item() in self.idx2char)
            cleaned_str = decoded_str.lstrip('0') or '0'
            batch_decoded.append(cleaned_str)
        
        return batch_decoded  
    
    def __len__(self):
        return len(self.questions)
    
    def __getitem__(self, idx):
        return self.encode_string(self.questions[idx], (2*self.max_digits) + 1), self.encode_string(self.answers[idx], (2*self.max_digits) + 1)

import torch
import torch.nn as nn
import pytorch_lightning as pl
import random
from colorama import Fore

class ArithmeticSeq2Seq(pl.LightningModule):
    def __init__(
        self,
        dataset,
        embed_dim=128,
        hidden_dim=128,
        num_layers=1,
        lr=0.001,
        teacher_forcing_ratio=0.5,
    ):
        super().__init__()
        self.dataset = dataset
        self.save_hyperparameters()

        # Minimal toy LSTM encoder/decoder
        self.embedding = nn.Embedding(len(dataset.CHARS), embed_dim)
        self.encoder = nn.LSTM(embed_dim, hidden_dim, num_layers=num_layers, batch_first=True)
        self.decoder = nn.LSTM(embed_dim, hidden_dim, num_layers=num_layers, batch_first=True)
        self.fc = nn.Linear(hidden_dim, len(dataset.CHARS))

        self.loss_fn = nn.CrossEntropyLoss()
        self.lr = lr
        self.teacher_forcing_ratio = teacher_forcing_ratio

        # We'll store samples from val step here
        self.val_samples = []

    def forward(self, src, trg):
        """
        A toy seq2seq approach.
        src: (batch_size, src_len)
        trg: (batch_size, trg_len)
        returns a tensor of shape (batch_size, trg_len, vocab_size)
        """
        batch_size, trg_len = trg.shape

        # === 1) Reverse the source tokens ===
        # flips dimension 1 (the sequence length) so the model sees the question from last digit to first.
        src_reversed = torch.flip(src, dims=[1])

        # === 2) Encode reversed src ===
        embedded_src = self.embedding(src_reversed)      # (B, src_len, E)
        _, (h, c) = self.encoder(embedded_src)           # (h, c) => (num_layers, B, H)

        # === 3) Prepare output placeholder ===
        outputs = torch.zeros(batch_size, trg_len, len(self.dataset.CHARS), device=self.device)

        # === 4) Decode step by step ===
        input_token = trg[:, 0]  # first token in target
        for t in range(1, trg_len):
            embedded_t = self.embedding(input_token).unsqueeze(1)  # (B,1,E)
            decoder_out, (h, c) = self.decoder(embedded_t, (h, c))  # (B,1,H)
            prediction = self.fc(decoder_out.squeeze(1))           # (B,vocab_size)
            outputs[:, t, :] = prediction

            # teacher forcing
            teacher_force = random.random() < self.teacher_forcing_ratio
            top1 = prediction.argmax(dim=1)
            input_token = trg[:, t] if teacher_force else top1

        return outputs

    def training_step(self, batch, batch_idx):
        src, trg = batch
        predictions = self(src, trg)
        predictions = predictions[:, 1:, :].reshape(-1, len(self.dataset.CHARS))
        targets     = trg[:, 1:].reshape(-1)

        loss = self.loss_fn(predictions, targets)
        self.log("train_loss", loss, prog_bar=True)
        return loss

    def validation_step(self, batch, batch_idx):
        """
        1) Compute val loss.
        2) Pick 1 random sample from this batch.
        3) Do a teacher_forcing=0 decode for demonstration.
        4) Store the sample in self.val_samples.
        """
        src, trg = batch
        predictions = self(src, trg)
        predictions = predictions[:, 1:, :].reshape(-1, len(self.dataset.CHARS))
        targets     = trg[:, 1:].reshape(-1)

        loss = self.loss_fn(predictions, targets)
        self.log("val_loss", loss, prog_bar=True)

        # Pick a random sample for demonstration
        sample_idx = random.randint(0, src.size(0) - 1)

        # We'll decode again with teacher_forcing=0
        saved_tfr = self.teacher_forcing_ratio
        self.teacher_forcing_ratio = 0.0
        with torch.no_grad():
            preds = self(src[sample_idx:sample_idx+1], trg[sample_idx:sample_idx+1])
        self.teacher_forcing_ratio = saved_tfr

        best_tokens = preds.argmax(dim=2)  # shape (1, trg_len)
        input_str   = self.dataset.decode_string(src[sample_idx:sample_idx+1])
        pred_str    = self.dataset.decode_string(best_tokens)
        true_str    = self.dataset.decode_string(trg[sample_idx:sample_idx+1])

        color = Fore.GREEN if pred_str[0] == true_str[0] else Fore.RED

        # Store the sample in self.val_samples => We'll print them in on_validation_epoch_end
        self.val_samples.append({
            "batch_idx":  batch_idx,
            "color":      color,
            "input":      input_str[0],
            "prediction": pred_str[0],
            "target":     true_str[0],
        })

        return loss

    def on_validation_epoch_end(self):
        """
        Called after all val batches in an epoch are processed.
        We'll only print if current_epoch+1 is multiple of 15 here, just as an example.
        """
        if self.trainer.sanity_checking:
            # skip printing if in sanity check
            self.val_samples.clear()
            return

        current_epoch = self.current_epoch + 1
        if current_epoch % 15 == 0:
            for sample in self.val_samples:
                print(
                    f"{sample['color']}[Val Epoch {current_epoch}, Batch {sample['batch_idx']}] "
                    f"Input: {sample['input']} | Prediction: {sample['prediction']} | Target: {sample['target']}"
                )

        self.val_samples.clear()

    def inference_decode(self, question_str, max_len=9):
        """
        Decodes a single query string (question_str) into a predicted answer string,
        with no teacher forcing, step-by-step, reversing the question input before encode.
        """
        self.eval()
        with torch.no_grad():
            # Encode the question
            # 1) Convert question string to tensor
            src = self.dataset.encode_string(question_str, max_len).unsqueeze(0).to(self.device)
            
            # 2) Reverse the question tokens
            src_reversed = torch.flip(src, dims=[1])

            # 3) Run encoder
            embedded_src = self.embedding(src_reversed)
            _, (h, c) = self.encoder(embedded_src)
            
            # 4) Decoder loop with no teacher forcing
            outputs = torch.zeros(1, max_len, len(self.dataset.CHARS), device=self.device)
            
            # We'll pick '0' as our initial token
            start_idx = self.dataset.char2idx['0']
            input_token = torch.tensor([start_idx], device=self.device)
            
            for t in range(1, max_len):
                embedded_t = self.embedding(input_token).unsqueeze(1)
                decoder_out, (h, c) = self.decoder(embedded_t, (h, c))
                prediction = self.fc(decoder_out.squeeze(1))
                outputs[:, t, :] = prediction
                # pick top token
                input_token = prediction.argmax(dim=1)
            
            # Convert the best tokens to a string
            best_tokens = outputs.argmax(dim=2)  # shape: (1, max_len)
            pred_str = self.dataset.decode_string(best_tokens)
            return pred_str[0]  # single decoded string

    def configure_optimizers(self):
        return torch.optim.Adam(self.parameters(), lr=self.lr)

from colorama import Fore

def query_model(model, dataset, expression, max_len=9):
    """
    Given a query expression like '234+12', this function:
    1) Evaluates the ground truth with Python's eval
    2) Decodes the model's prediction using model.inference_decode
    3) Prints color-coded result (green if correct, red otherwise)
    4) Returns (ground_truth, predicted_answer)
    """
    # 1) Evaluate ground truth
    try:
        ground_truth = str(eval(expression))
    except:
        ground_truth = "Error"
    
    # 2) Get the model prediction via inference decode
    predicted_answer = model.inference_decode(expression, max_len=max_len)
    
    # 3) Determine color
    color = Fore.GREEN if predicted_answer.strip() == ground_truth.strip() else Fore.RED
    
    # 4) Print result
    print(Fore.WHITE + f"Expression: {expression}")
    print(color + f"Truth: {ground_truth} | Model Prediction: {predicted_answer}")
    
    return ground_truth, predicted_answer

import torch
import torch.nn as nn
import pytorch_lightning as pl
import torchmetrics.functional as F

class LSTMClassifier(pl.LightningModule):
    def __init__(self, vocab_size, embed_dim=128, hidden_dim=128, num_layers=1, output_dim=4, lr=0.001):
        """
        A text classification model using LSTM.
        
        Args:
            vocab_size (int): Size of the vocabulary.
            embed_dim (int): Dimension of the word embeddings.
            hidden_dim (int): Number of hidden units in the LSTM.
            num_layers (int): Number of LSTM layers (default=1).
            output_dim (int): Number of output classes.
            lr (float): Learning rate (default=0.001).
        """
        super().__init__()
        self.save_hyperparameters()

        # Embedding layer
        self.embedding = nn.Embedding(vocab_size, embed_dim)

        # LSTM layer(s) with variable number of layers
        self.lstm = nn.LSTM(embed_dim, hidden_dim, num_layers=num_layers, batch_first=True)

        # Fully connected output layer
        self.fc = nn.Linear(hidden_dim, output_dim)

        # Loss function
        self.loss_fn = nn.CrossEntropyLoss()
        self.lr = lr

    def forward(self, x):
        """
        Forward pass through the LSTMClassifier.

        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, seq_len).
        
        Returns:
            torch.Tensor: Output logits of shape (batch_size, output_dim).
        """
        embedded = self.embedding(x)  # (batch_size, seq_len, embed_dim)
        lstm_out, _ = self.lstm(embedded)  # (batch_size, seq_len, hidden_dim)

        # Take the last hidden state (last time step)
        last_hidden_state = lstm_out[:, -1, :]  # (batch_size, hidden_dim)

        return self.fc(last_hidden_state)  # (batch_size, output_dim)

    def training_step(self, batch, batch_idx):
        x, y = batch
        logits = self(x)
        loss = self.loss_fn(logits, y)

        # Compute accuracy
        acc = F.accuracy(logits.softmax(dim=-1), y, task="multiclass", num_classes=self.hparams.output_dim)

        # Log loss and accuracy
        self.log("train_loss", loss, prog_bar=True, on_step=False, on_epoch=True)
        self.log("train_acc", acc, prog_bar=True, on_step=False, on_epoch=True)

        return loss

    def validation_step(self, batch, batch_idx):
        x, y = batch
        logits = self(x)
        loss = self.loss_fn(logits, y)

        # Compute accuracy
        acc = F.accuracy(logits.softmax(dim=-1), y, task="multiclass", num_classes=self.hparams.output_dim)

        # Log loss and accuracy
        self.log("val_loss", loss, prog_bar=True, on_step=False, on_epoch=True)
        self.log("val_acc", acc, prog_bar=True, on_step=False, on_epoch=True)

        return loss

    def configure_optimizers(self):
        """
        Uses AdamW optimizer with weight decay for better regularization.
        """
        optimizer = torch.optim.AdamW(self.parameters(), lr=self.lr, weight_decay=.01)
        return optimizer

import pandas as pd

def process_csv_logger(csv_path: str) -> pd.DataFrame:
    """
    Reads a CSV logger file and processes it into a rectangular DataFrame with one row per epoch.
    
    The CSV must contain at least the columns "epoch" and "step", along with other metric columns.
    For each (epoch, metric) combination, if there are multiple entries, the row with the highest step is kept.
    
    Parameters:
        csv_path (str): The path to the CSV file.
        
    Returns:
        pd.DataFrame: A wide DataFrame with one row per epoch and columns for each metric.
    """
    # Read the CSV file into a DataFrame.
    df = pd.read_csv(csv_path)
    
    # Melt the DataFrame into long format.
    # Each row will have: epoch, step, metric (column name), and its corresponding value.
    long_df = df.melt(id_vars=['epoch', 'step'], var_name='metric', value_name='value')
    
    # Drop rows where value is missing.
    long_df = long_df.dropna(subset=['value'])
    
    # For each epoch and metric, keep the row with the largest step.
    idx = long_df.groupby(['epoch', 'metric'])['step'].idxmax()
    long_df_max = long_df.loc[idx]
    
    # Pivot the long DataFrame back to wide format: one row per epoch, columns for each metric.
    wide_df = long_df_max.pivot(index='epoch', columns='metric', values='value').reset_index()
    
    return wide_df
